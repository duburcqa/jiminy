183  def _get_colorized_urdf(urdf_path, rgb, custom_mesh_search_path=None):
184  """
185  @brief Generate a unique colorized URDF.
186 
187  @remark Multiple identical URDF model of different colors can be
188  loaded in Gepetto-viewer this way.
189 
190  @param[in] urdf_path Full path of the URDF file
191  @param[in] rgb RGB code defining the color of the model. It is the same for each link.
192 
193  @return Full path of the colorized URDF file.
194  """
195 
196  color_string = "%.3f_%.3f_%.3f_1.0" % rgb
197  color_tag = "<color rgba=\"%.3f %.3f %.3f 1.0\"" % rgb # don't close tag with '>', in order to handle <color/> and <color></color>
198  colorized_tmp_path = os.path.join("/tmp", "colorized_urdf_rgba_" + color_string)
199  colorized_urdf_path = os.path.join(colorized_tmp_path, os.path.basename(urdf_path))
200  if not os.path.exists(colorized_tmp_path):
201  os.makedirs(colorized_tmp_path)
202 
203  with open(urdf_path, 'r') as urdf_file: colorized_contents = urdf_file.read()
204 
205  for mesh_fullpath in re.findall('<mesh filename="(.*)"', colorized_contents):
206  colorized_mesh_fullpath = os.path.join(colorized_tmp_path, mesh_fullpath[1:])
207  colorized_mesh_path = os.path.dirname(colorized_mesh_fullpath)
208  if not os.access(colorized_mesh_path, os.F_OK):
209  os.makedirs(colorized_mesh_path)
210  shutil.copy2(mesh_fullpath, colorized_mesh_fullpath)
211  colorized_contents = colorized_contents.replace('"' + mesh_fullpath + '"',
212  '"' + colorized_mesh_fullpath + '"', 1)
213  colorized_contents = re.sub("<color rgba=\"[\d. ]*\"", color_tag, colorized_contents)
214 
215  with open(colorized_urdf_path, 'w') as colorized_urdf_file:
216  colorized_urdf_file.write(colorized_contents)
217 
218  return colorized_urdf_path

316  def setCameraTransform(self, translation, rotation):
317  # translation : [Px, Py, Pz],
318  # rotation : [Roll, Pitch, Yaw]
319 
320  R_pnc = rpyToMatrix(np.array(rotation))
321  if Viewer.backend == 'gepetto-gui':
322  T_pnc = np.array(translation)

325  else:
326  import meshcat.transformations as tf
327  # Transformation of the camera object

329  self._client.viewer["/Cameras/default/rotated/<object>"].set_transform(T_meshcat)
330  # Orientation of the camera object
331  Q_pnc = Quaternion(R_pnc).coeffs()
332  Q_meshcat = np.roll(Q_pnc, shift=1)

384  def display(self, evolution_robot, speed_ratio, xyz_offset=None):
385  t = [s.t for s in evolution_robot]
386  i = 0
387  init_time = time.time()
388  while i < len(evolution_robot):
389  s = evolution_robot[i]
390  if (xyz_offset is not None):
391  q = s.q.copy() # Make sure to use a copy to avoid altering the original data
392  q[:3] += xyz_offset
393  else:
394  q = s.q
395  with Viewer._lock: # It is necessary to use Lock since corbaserver does not support multiple connection simultaneously.
396  self._rb.display(q)
397  t_simu = (time.time() - init_time) * speed_ratio
398  i = bisect_right(t, t_simu)
399  if t_simu < s.t:
400  time.sleep(s.t - t_simu)

403 def play_trajectories(trajectory_data, xyz_offset=None, urdf_rgba=None, speed_ratio=1.0,
404  backend='gepetto-gui', window_name='python-pinocchio', scene_name='world'):
405  """!
406  @brief Display robot evolution in Gepetto-viewer at stable speed.
407 
408  @remark The speed is independent of the machine, and more
409  specifically of CPU power.
410 
411  @param[in] trajectory_data Trajectory dictionary with keys:
412  'evolution_robot': list of State object of increasing time
413  'jiminy_model': Jiminy model (None if omitted)
414  'use_theoretical_model': whether the theoretical or actual model must be used
415  @param[in] xyz_offset Constant translation of the root joint in world frame (1D numpy array)
416  @param[in] urdf_rgba RGBA code defining the color of the model. It is the same for each link.
417  Optional: Original colors of each link. No alpha.
418  @param[in] speed_ratio Speed ratio of the simulation
419  @param[in] window_name Name of the Gepetto-viewer's window in which to display the robot.
420  Optional: Common default name if omitted.
421  @param[in] scene_name Name of the Gepetto-viewer's scene in which to display the robot.
422  Optional: Common default name if omitted.
423  """
424 
425  # Load robots in gepetto viewer
426  robots = []
427  for i in range(len(trajectory_data)):
428  jiminy_model = trajectory_data[i]['jiminy_model']
429  use_theoretical_model = trajectory_data[i]['use_theoretical_model']
430  robot = Viewer(jiminy_model, use_theoretical_model=use_theoretical_model,
431  urdf_rgba=urdf_rgba[i] if urdf_rgba is not None else None, robot_index=i,
432  backend=backend, window_name=window_name, scene_name=scene_name)
433 
434  if (xyz_offset is not None and xyz_offset[i] is not None):
435  q = trajectory_data[i]['evolution_robot'][0].q.copy() # Make sure to use a copy to avoid altering the original data
436  q[:3] += xyz_offset[i]
437  else:
438  q = trajectory_data[i]['evolution_robot'][0].q
439  robot._rb.display(q)
440 
441  robots.append(robot)
442 
443  if (xyz_offset is None):
444  xyz_offset = len(trajectory_data) * (None,)
445 
446  threads = []
447  for i in range(len(trajectory_data)):
448  threads.append(Thread(target=robots[i].display,
449  args=(trajectory_data[i]['evolution_robot'],
450  speed_ratio, xyz_offset[i])))
451  for i in range(len(trajectory_data)):
452  threads[i].start()
453  for i in range(len(trajectory_data)):
454  threads[i].join()
455 
456  Viewer.close()
