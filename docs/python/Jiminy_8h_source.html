346  static bp::object getItemSplit(sensorsDataMap_t & self,
347  std::string const & sensorType,
348  std::string const & sensorName)
349  {
350  try
351  {
352  auto const & sensorDataType = self.at(sensorType);
353  auto sensorIt = std::find_if(sensorDataType.begin(),
354  sensorDataType.end(),
355  [&sensorName](auto const & element)
356  {
357  return element.first == sensorName;
358  });
359  bp::handle<> valuePy(getNumpyReferenceFromEigenVector(*sensorIt->second));
360  return bp::object(valuePy);
361  }
362  catch (...)
363  {
364  PyErr_SetString(PyExc_KeyError, "The key does not exist.");
365  return bp::object();
366  }

369  static matrixN_t getSub(sensorsDataMap_t & self,
370  std::string const & sensorType)
371  {
372  // Extract the encoder data matrix
373  auto const & sensorsDataType = self.at(sensorType);
374  matrixN_t data;
375  auto sensorDataIt = sensorsDataType.begin();
376  vectorN_t const & sensorValue = *sensorDataIt->second;
377  data.resize(sensorValue.size(), sensorsDataType.size());
378  data.col(0) = sensorValue;
379  ++sensorDataIt;
380  for (uint32_t i = 1; i<sensorsDataType.size(); ++i)
381  {
382  data.col(i) = *sensorDataIt->second;
383  ++sensorDataIt;
384  }
385  return data;
386  }

388  static bool contains(sensorsDataMap_t & self,
389  bp::tuple const & sensorInfo)
390  {
391  std::string sensorType = bp::extract<std::string>(sensorInfo[0]);
392  std::string sensorName = bp::extract<std::string>(sensorInfo[1]);
393  auto const & sensorDataType = self.find(sensorType);
394  if (sensorDataType != self.end())
395  {
396  auto it = std::find_if(sensorDataType->second.begin(),
397  sensorDataType->second.end(),
398  [&sensorName](auto const & element)
399  {
400  return element.first == sensorName;
401  });
402  if (it != sensorDataType->second.end())
403  {
404  return true;
405  }
406  }
407  return false;
408  }

1187  static bp::tuple formatLog(std::vector<std::string> const & header,
1188  std::vector<float32_t> const & timestamps,
1189  std::vector<std::vector<int32_t> > & intData,
1190  std::vector<std::vector<float32_t> > & floatData,
1191  bool const & clear_memory = true)
1192  {
1193  bp::dict constants;
1194  bp::dict data;
1195 
1196  // Get constants
1197  uint32_t lastConstantId = std::distance(header.begin(), std::find(header.begin(), header.end(), START_COLUMNS));
1198  for (uint32_t i = 1; i < lastConstantId; i++)
1199  {
1200  int32_t delimiter = header[i].find("=");
1201  constants[header[i].substr(0, delimiter)] = header[i].substr(delimiter + 1);
1202  }
1203 
1204  // Get Global.Time
1205  Eigen::Ref<Eigen::Matrix<float32_t, Eigen::Dynamic, 1> const> timeBuffer =
1206  Eigen::Matrix<float32_t, Eigen::Dynamic, 1>::Map(
1207  timestamps.data(), timestamps.size());
1208  PyObject * valuePyTime(getNumpyReferenceFromEigenVector(timeBuffer));
1209  data[header[lastConstantId + 1]] = bp::object(bp::handle<>(PyArray_FROM_OF(valuePyTime, NPY_ARRAY_ENSURECOPY)));
1210  Py_XDECREF(valuePyTime);
1211 
1212  // Get intergers
1213  Eigen::Matrix<int32_t, Eigen::Dynamic, Eigen::Dynamic> intDataMatrix;
1214  intDataMatrix.resize(timestamps.size(), intData[0].size());
1215  for (uint32_t i=0; i<intData.size(); i++)
1216  {
1217  intDataMatrix.row(i) = Eigen::Matrix<int32_t, 1, Eigen::Dynamic>::Map(
1218  intData[i].data(), intData[0].size());
1219  }
1220  if (clear_memory)
1221  {
1222  intData.clear();
1223  }
1224 
1225  for (uint32_t i=0; i<intData[0].size(); i++)
1226  {
1227  PyObject * valuePyInt(getNumpyReferenceFromEigenVector(intDataMatrix.col(i)));
1228  std::string const & header_i = header[i + (lastConstantId + 1) + 1];
1229  // One must make copies with PyArray_FROM_OF instead of using raw pointer for floatDataMatrix
1230  // and setting NPY_ARRAY_OWNDATA because otherwise Python is not able to free the memory
1231  // associated to each columns independently.
1232  // Moreover, one must decrease manually the counter reference for some reason...
1233  data[header_i] = bp::object(bp::handle<>(PyArray_FROM_OF(valuePyInt, NPY_ARRAY_ENSURECOPY)));
1234  Py_XDECREF(valuePyInt);
1235  }
1236 
1237  // Get floats
1238  Eigen::Matrix<float32_t, Eigen::Dynamic, Eigen::Dynamic> floatDataMatrix;
1239  floatDataMatrix.resize(timestamps.size(), floatData[0].size());
1240  for (uint32_t i=0; i<floatData.size(); i++)
1241  {
1242  floatDataMatrix.row(i) = Eigen::Matrix<float32_t, 1, Eigen::Dynamic>::Map(
1243  floatData[i].data(), floatData[0].size());
1244  }
1245  if (clear_memory)
1246  {
1247  floatData.clear();
1248  }
1249 
1250  for (uint32_t i=0; i<floatData[0].size(); i++)
1251  {
1252  PyObject * valuePyFloat(getNumpyReferenceFromEigenVector(floatDataMatrix.col(i)));
1253  std::string const & header_i = header[i + (lastConstantId + 1) + 1 + intData[0].size()];
