18 def _worker(remote, parent_remote, env_fn_wrapper, lock=None):
19  """
20  @brief Worker for each subprocess of SubprocVecEnvLock.
21 
22  @details It enables the use of a threading.Lock. It is the only
23  difference with the implementation provided with the
24  class `SubprocVecEnv` of Gym OpenAI.
25 
26  @remark `remote` and `parent_remote` are returned by the method `Pipe` of
27  `multiprocessing`. It is a pair of connection objects connected
28  by a pipe which by default is duplex (two-way). See
29  `multiprocessing` documentation for more information.
30 
31  This is a hidden function that is not automatically imported
32  using 'from gym_jiminy import *'.
33 
34  @param[in] remote Child remote
35  @param[in] parent_remote Parent remote
36  @param[in] env_fn_wrapper Gym environment
37  @param[in] lock threading.Lock object. Optional: None by default
38  """
39  parent_remote.close()
40  env = env_fn_wrapper.var()
41  while True:
42  try:
43  cmd, data = remote.recv()
44  if cmd == 'step':
45  observation, reward, done, info = env.step(data)
46  if done:
47  # save final observation where user can get it, then reset
48  info['terminal_observation'] = observation
49  observation = env.reset()
50  remote.send((observation, reward, done, info))
51  elif cmd == 'reset':
52  observation = env.reset()
53  remote.send(observation)
54  elif cmd == 'render':
55  remote.send(env.render(*data[0], lock=lock, **data[1]))
56  elif cmd == 'close':
57  remote.close()
58  break
59  elif cmd == 'get_spaces':
60  remote.send((env.observation_space, env.action_space))
61  elif cmd == 'env_method':
62  method = getattr(env, data[0])
63  remote.send(method(*data[1], **data[2]))
64  elif cmd == 'get_attr':
65  remote.send(getattr(env, data))
66  elif cmd == 'set_attr':
67  remote.send(setattr(env, data[0], data[1]))
68  else:
69  raise NotImplementedError
70  except EOFError:
71  break

74 class SubprocVecEnvLock(SubprocVecEnv):
75  """
76  @brief Creates a multiprocess vectorized wrapper for multiple environments,
77  distributing each environment to its own process, allowing
78  significant speed up when the environment is computationally complex.
79 
80  @details It features a unique threading.Lock and uses it to enable parallel
81  rendering in Gepetto-viewer. It is the only difference with the
82  based class `SubprocVecEnv` provided by Gym OpenAI.
83 
84  @warning For performance reasons, if your environment is not IO bound, the
85  number of environments should not exceed the number of logical cores
86  on your CPU.
87  """
88  def __init__(self, env_fns, start_method=None):
89  """
90  @brief Constructor
91 
92  @warning Only 'forkserver' and 'spawn' start methods are thread-safe, which is
93  important when TensorFlow sessions or other non thread-safe libraries
94  are used in the parent.
95  However, compared to 'fork' they incur a small start-up cost and have
96  restrictions on global variables. With those methods, users must wrap
97  the code in an ``if __name__ == "__main__":``
98  For more information, see the multiprocessing documentation.
99 
100  @param[in] env_fns List of Gym Environments to run in subprocesses
101  @param[in] start_method Method used to start the subprocesses. Must be one of the
102  methods returned by multiprocessing.get_all_start_methods().
103  Optional: Defaults to 'fork' on available platforms, and 'spawn' otherwise.
104 
105  @return Instance of SubprocVecEnvLock.
106  """
107  global lock
108 
109  self.waiting = False
110  self.closed = False
111  n_envs = len(env_fns)
112 
113  if start_method is None:
114  # Fork is not a thread safe method (see issue #217)
115  # but is more user friendly (does not require to wrap the code in
116  # a `if __name__ == "__main__":`)
117  fork_available = 'fork' in multiprocessing.get_all_start_methods()
118  start_method = 'fork' if fork_available else 'spawn'
119  ctx = multiprocessing.get_context(start_method)
120 
121  self.remotes, self.work_remotes = zip(*[ctx.Pipe() for _ in range(n_envs)])
122  self.processes = []
123  for work_remote, remote, env_fn in zip(self.work_remotes, self.remotes, env_fns):
124  args = (work_remote, remote, CloudpickleWrapper(env_fn), lock)
125  # daemon=True: if the main process crashes, we should not cause things to hang
126  process = ctx.Process(target=_worker, args=args, daemon=True)
127  process.start()
128  self.processes.append(process)
129  work_remote.close()
