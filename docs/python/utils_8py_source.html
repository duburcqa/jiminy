126 def smoothing_filter(time_in, val_in, time_out=None, relabel=None, params=None):
127  """
128  @brief Smoothing filter with relabeling and resampling features.
129 
130  @details It supports evenly sampled multidimensional input signal.
131  Relabeling can be used to infer the value of samples at
132  time steps before and after the explicitly provided samples.
133  As a reminder, relabeling is a generalization of periodicity.
134 
135  @param[in] time_in Time steps of the input signal (1D numpy array)
136  @param[in] val_in Sampled values of the input signal
137  (2D numpy array: row = sample, column = time)
138  @param[in] time_out Time steps of the output signal (1D numpy array)
139  @param[in] relabel Relabeling matrix (identity for periodic signals)
140  Optional: Disable if omitted
141  @param[in] params Parameters of the filter. Dictionary with keys:
142  'mixing_ratio_1': Relative time at the begining of the signal
143  during the output signal corresponds to a
144  linear mixing over time of the filtered and
145  original signal. (only used if relabel is omitted)
146  'mixing_ratio_2': Relative time at the end of the signal
147  during the output signal corresponds to a
148  linear mixing over time of the filtered and
149  original signal. (only used if relabel is omitted)
150  'smoothness'[0]: Smoothing factor to filter the begining of the signal
151  (only used if relabel is omitted)
152  'smoothness'[1]: Smoothing factor to filter the end of the signal
153  (only used if relabel is omitted)
154  'smoothness'[2]: Smoothing factor to filter the middle part of the signal
155 
156  @return Filtered signal (2D numpy array: row = sample, column = time)
157  """
158  if time_out is None:
159  time_out = time_in
160  if params is None:
161  params = dict()
162  params['mixing_ratio_1'] = 0.12
163  params['mixing_ratio_2'] = 0.04
164  params['smoothness'] = [0.0,0.0,0.0]
165  params['smoothness'][0] = 5e-3
166  params['smoothness'][1] = 5e-3
167  params['smoothness'][2] = 3e-3
168 
169  if relabel is None:
170  mix_fit = [None,None,None]
171  mix_fit[0] = lambda t: 0.5*(1+np.sin(1/params['mixing_ratio_1']*((t-time_in[0])/(time_in[-1]-time_in[0]))*np.pi-np.pi/2))
172  mix_fit[1] = lambda t: 0.5*(1+np.sin(1/params['mixing_ratio_2']*((t-(1-params['mixing_ratio_2'])*time_in[-1])/(time_in[-1]-time_in[0]))*np.pi+np.pi/2))
173  mix_fit[2] = lambda t: 1
174 
175  val_fit = []
176  for jj in range(val_in.shape[0]):
177  val_fit_jj = []
178  for kk in range(len(params['smoothness'])):
179  val_fit_jj.append(UnivariateSpline(time_in, val_in[jj], s=params['smoothness'][kk]))
180  val_fit.append(val_fit_jj)
181 
182  time_out_mixing = [None, None, None]
183  time_out_mixing_ind = [None, None, None]
184  time_out_mixing_ind[0] = time_out < time_out[-1]*params['mixing_ratio_1']
185  time_out_mixing[0] = time_out[time_out_mixing_ind[0]]
186  time_out_mixing_ind[1] = time_out > time_out[-1]*(1-params['mixing_ratio_2'])
187  time_out_mixing[1] = time_out[time_out_mixing_ind[1]]
188  time_out_mixing_ind[2] = np.logical_and(np.logical_not(time_out_mixing_ind[0]), np.logical_not(time_out_mixing_ind[1]))
189  time_out_mixing[2] = time_out[time_out_mixing_ind[2]]
190 
191  val_out = np.zeros((val_in.shape[0],len(time_out)))
192  for jj in range(val_in.shape[0]):
193  for kk in range(len(time_out_mixing)):
194  val_out[jj,time_out_mixing_ind[kk]] = \
195  (1 - mix_fit[kk](time_out_mixing[kk])) * val_fit[jj][kk](time_out_mixing[kk]) + \
196  mix_fit[kk](time_out_mixing[kk]) * val_fit[jj][-1](time_out_mixing[kk])
197  else:
198  time_tmp = np.concatenate([time_in[:-1]-time_in[-1],time_in,time_in[1:]+time_in[-1]])
199  val_in_tmp = np.concatenate([relabel.dot(val_in[:,:-1]),val_in,relabel.dot(val_in[:,1:])], axis=1)
200  val_out = np.zeros((val_in.shape[0],len(time_out)))
201  for jj in range(val_in_tmp.shape[0]):
202  f = UnivariateSpline(time_tmp, val_in_tmp[jj], s=params['smoothness'][-1])
203  val_out[jj] = f(time_out)
204 
205  return val_out

428  """

432  colorized_tmp_path = os.path.join("/tmp", "colorized_urdf_rgba_" + color_string)

442  if not os.access(colorized_mesh_path, os.F_OK):
443  os.makedirs(colorized_mesh_path)

445  colorized_contents = colorized_contents.replace('"' + mesh_fullpath + '"',

448

450  colorized_urdf_file.write(colorized_contents)
