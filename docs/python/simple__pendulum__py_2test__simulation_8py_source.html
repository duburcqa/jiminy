151 def updateState(model, q, v, sensor_data):
152  # Get dcm from current state
153  pnc.forwardKinematics(model.pinocchio_model_th, model.pinocchio_data_th, q, v)
154  comOut = pnc.centerOfMass(model.pinocchio_model_th, model.pinocchio_data_th, q, v)
155  vcomOut = model.pinocchio_data_th.vcom[0]
156  dcmOut = comOut + vcomOut / omega
157  # Create zmp from forces
158  forces = np.asarray(sensor_data[ForceSensor.type])
159  newWrench = pnc.Force.Zero()
160  for i,name in enumerate(contacts):
161  update_frame(model.pinocchio_model_th, model.pinocchio_data_th, name)
162  placement = get_frame_placement(model.pinocchio_model_th, model.pinocchio_data_th, name)
163  wrench = pnc.Force(np.concatenate([[0.0, 0.0, forces[2, i]], np.zeros(3)]).T)
164  newWrench += placement.act(wrench)
165  totalWrenchOut = newWrench
166  if totalWrench.linear[2] != 0:
167  zmpOut = [-totalWrenchOut.angular[1] / totalWrenchOut.linear[2],
168  totalWrenchOut.angular[0] / totalWrenchOut.linear[2]]
169  else:
170  zmpOut = zmp_log
171  return comOut, vcomOut, dcmOut, zmpOut, totalWrenchOut

173 def computeCommand(t, q, v, sensor_data, u):
174  global com, dcm, zmp, zmp_cmd, totalWrench, qi, dqi, ddqi, t_1, u_1, integral_
175 
176  # Get trajectory
177  i = int(t * 1e3) + 1
178  if t > taux :
179  p = paux
180  else:

182  z = zmpRef[i] + p

207  # Update targets at HLC frequency
208  if int(t * 1e3) % int(1e3 / fHLC) == 0:

210  # Compute zmp command (DCM control)
211  if args.targetsFB:

222  t_1 = t
223  zmp_cmd = z - (1 + Kpdcm / omega) * (d - di[axisCom]) + (Kddcm / omega) * (z - zi[axisCom]) - Kidcm / omega * integral_

227  # Compute joint acceleration from ZMP command
228  # Zmp-> com admittance -> com acceleration

230  # Com acceleration -> joint acceleration

232 
233  # Compute joint torque from joint acceleration (ID)
234  if acceleration_control:
235  u_1 = m * (l**2) * ((ax / (l * np.cos(q[iPos]))) + (v[iVel]**2) * np.tan(q[iPos]) - g * np.sin(q[iPos]) / l)
236 
237  # Send last joint torque command
238  if acceleration_control:
239  u[0] = u_1
240  # Integrate last joint acceleration + position control
241  elif position_control:
242  dqi[iVel] = dqi[iVel] + ddqi[iVel] * 1e-3
243  qi[iPos] = qi[iPos] + dqi[iVel] * 1e-3
244  u[0] = -(Kp * (q[iPos] - qi[iPos]) + Kd * (v[iVel] - dqi[iVel]))
245 
246  # Update logs (only the value stored by the registered variables using [:])
247  zmp_cmd_log[:] = zmp_cmd
248  state_target_log[0], state_target_log[1] = qi[iPos], dqi[iVel]
