21 class EngineAsynchronous(object):
22  """
23  @brief Wrapper of Jiminy enabling to update of the command and run simulation
24  steps asynchronously. Convenient helper methods are available to set the
25  seed of the simulation, reset it, and display it.
26 
27  @details The method `action` is used to update the command, which
28  is kept in memory until `action` is called again. On its side, the method
29  `step` without argument is used to run simulation steps. The method `step`
30  has an optional argument `dt_desired` to specify the number of simulation
31  steps to perform at once.
32 
33  @remark This class can be used for synchronous purpose. In such a case, one has
34  to call the method `step` specifying the optional argument `action_next`.
35  """
36  def __init__(self, model, viewer_backend=None, viewer_use_theoretical_model=False):
37  """
38  @brief Constructor
39 
40  @param[in] model Jiminy model properly setup (eg sensors already added)
41 
42  @return Instance of the wrapper.
43  """
44 
45  # Make sure that the sensors have already been added to the model !
46  self._sensors_types = model.get_sensors_options().keys()
47  self._state = np.zeros((model.nx,))
48  self._observation = OrderedDict((sensor_type,[]) for sensor_type in self._sensors_types)
49  self._action = np.zeros((len(model.motors_names),))
50 
51  # Instantiate the Jiminy controller
52  self._controller = jiminy.ControllerFunctor(self._send_command, self._internal_dynamics)
53  self._controller.initialize(model)
54 
55  # Instantiate the Jiminy engine (model and controller are pass-by-reference)
56  self._engine = jiminy.Engine()
57  self._engine.initialize(model, self._controller)
58 
59 
60  self.viewer_backend = viewer_backend
61  self.viewer_use_theoretical_model = viewer_use_theoretical_model
62  self._viewer = None
63 
64  self.reset()
65 
66  def _send_command(self, t, q, v, sensor_data, uCommand):
67  """
68  @brief This method implement the callback function required by Jiminy
69  Controller to get the command. In practice, it only updates a
70  variable shared between C++ and Python to the internal value
71  stored by this class.
72 
73  @remark This is a hidden function that is not listed as part of the
74  member methods of the class. It is not intended to be called
75  manually.
76  """
77  for sensor_type in self._sensors_types:
78  self._observation[sensor_type] = sensor_data[sensor_type]
79  uCommand[:] = self._action
80 
81  def _internal_dynamics(self, t, q, v, sensor_data, uCommand):
82  """
83  @brief This method implement the callback function required by Jiminy
84  Controller to get the internal dynamics. In practice, it does
85  nothing.
86 
87  @remark This is a hidden function that is not listed as part of the
88  member methods of the class. It is not intended to be called
89  manually.
90  """
91  pass

93  def seed(self, seed):
94  """
95  @brief Set the seed of the simulation and reset the simulation.
96 
97  @details The initial state is zero. Execute the method `reset` manually
98  afterward to specify a different initial state.
99 
100  @param[in] seed Desired seed (Unsigned integer 32 bits)
101  """
102  engine_options = self._engine.get_options()
103  engine_options["stepper"]["randomSeed"] = np.array(seed, dtype=np.dtype('uint32'))
104  self.reset(x0=None)
105  self._engine.set_options(engine_options)

107  def reset(self, x0=None):
108  """
109  @brief Reset the simulation.
110 
111  @details The initial state is zero. Execute the method `reset` manually
112  afterward to specify a different initial state.
113 
114  @param[in] x0 Desired initial state (2D numpy array: column vector)
115  """
116  if (x0 is None):
117  x0 = np.zeros((self._engine.model.nx, 1))
118  if (int(self._engine.reset(x0)) != 1):
119  raise ValueError("Reset of engine failed.")
120  self._state = x0.squeeze()

122  def step(self, action_next=None, dt_desired=-1):
123  """
124  @brief Run simulation steps.
125 
126  @details Even if Jiminy Engine performs several simulation steps
127  internally, this method only output the final state.
128 
129  @param[in] action_next Updated command (2D numpy array: column vector)
130  Optional: Use the value in the internal buffer otherwise
131  @param[in] dt_desired Simulation time difference between before and after the steps.
132  Optional: Perform a single integration step otherwise
133 
134  @return Final state of the simulation (2D numpy matrix: column vector ???)
135  """
136  if (action_next is not None):
137  self.action = action_next
138  self._state = None
139  return self._engine.step(dt_desired)

141  def render(self, return_rgb_array=False, lock=None):
142  """
143  @brief Render the current state of the simulation. One can display it
144  in Gepetto-viewer or return an RGB array.
145 
146  @remark Note that it supports parallel rendering, which means that one
147  can display multiple simulations in the same Gepetto-viewer
148  processus at the same time in different tabs.
149  Note that it returning an RGB array required Gepetto-viewer.
150 
151  @param[in] return_rgb_array Updated command (2D numpy array: column vector)
152  Optional: Use the value in the internal buffer otherwise
153  @param[in] lock Unique threading.Lock for every simulation
154  Optional: Only required for parallel rendering
155 
156  @return Low-resolution rendering as an RGB array (3D numpy array)
157  """
158  rgb_array = None
159 
160  if (lock is not None):
161  lock.acquire()
162  try:
163  # Instantiate the Gepetto model and viewer client if necessary
164  if (self._viewer is None):
165  scene_name = next(tempfile._get_candidate_names())
166  self._viewer = Viewer(self._engine.model,
167  backend=self.viewer_backend,
168  use_theoretical_model=self.viewer_use_theoretical_model,
169  window_name='jiminy', scene_name=scene_name)
170  self._viewer.setCameraTransform(translation=[0.0, 9.0, 2e-5],
171  rotation=[np.pi/2, 0.0, np.pi])
172 
173  # Refresh viewer
174  self._viewer.refresh()
175 
176  # return rgb array if needed
177  if return_rgb_array:
178  rgb_array = self._viewer.captureFrame()
179  except:
180  self._viewer = None
181  finally:
182  if (lock is not None):
183  lock.release()
184  return rgb_array
