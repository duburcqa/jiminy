22 class EngineAsynchronous(object):
23  """
24  @brief Wrapper of Jiminy enabling to update of the command and run simulation
25  steps asynchronously. Convenient helper methods are available to set the
26  seed of the simulation, reset it, and display it.
27 
28  @details The method `action` is used to update the command, which
29  is kept in memory until `action` is called again. On its side, the method
30  `step` without argument is used to run simulation steps. The method `step`
31  has an optional argument `dt_desired` to specify the number of simulation
32  steps to perform at once.
33 
34  @remark This class can be used for synchronous purpose. In such a case, one has
35  to call the method `step` specifying the optional argument `action_next`.
36  """
37  def __init__(self, model):
38  """
39  @brief Constructor
40 
41  @param[in] model Jiminy model properly setup (eg sensors already added)
42 
43  @return Instance of the wrapper.
44  """
45  # Make sure that the sensors have already been added to the model !
46 
47  self._sensors_types = model.get_sensors_options().keys()
48  self._state = np.zeros((model.nx, 1))
49  self._observation = OrderedDict((sensor_type,[]) for sensor_type in self._sensors_types)
50  self._action = np.zeros((len(model.motors_names), ))
51 
52  # Instantiate the Jiminy controller
53  self._controller = jiminy.ControllerFunctor(
54  self._send_command, self._internal_dynamics, len(self._sensors_types))
55  self._controller.initialize(model)
56 
57  # Instantiate the Jiminy engine (model and controller are pass-by-reference)
58  self._engine = jiminy.Engine()
59  self._engine.initialize(model, self._controller)

72  def _send_command(self, t, q, v, *args):
73  """
74  @brief This method implement the callback function required by Jiminy
75  Controller to get the command. In practice, it only updates a
76  variable shared between C++ and Python to the internal value
77  stored by this class.
78 
79  @remark This is a hidden function that is not listed as part of the
80  member methods of the class. It is not intended to be called
81  manually.
82  """
83  for k, sensor_type in enumerate(self._observation):
84  self._observation[sensor_type] = args[k]
85  uCommand = args[-1]
86  uCommand[:] = self._action

102  def seed(self, seed):
103  """
104  @brief Set the seed of the simulation and reset the simulation.
105 
106  @details The initial state is zero. Execute the method `reset` manually
107  afterward to specify a different initial state.
108 
109  @param[in] seed Desired seed (Unsigned integer 32 bits)
110  """

113  self.reset(x0=None)

117  def reset(self, x0=None):
118  """
119  @brief Reset the simulation.
120 
121  @details The initial state is zero. Execute the method `reset` manually
122  afterward to specify a different initial state.
123 
124  @param[in] x0 Desired initial state (2D numpy array: column vector)
125  """
126  if (x0 is None):
127  x0 = np.zeros((self._engine.model.nx, 1))
128  if (int(self._engine.reset(x0)) != 1):
129  raise ValueError("Reset of engine failed.")

133  def step(self, action_next=None, dt_desired=-1):
134  """
135  @brief Run simulation steps.
136 
137  @details Even if Jiminy Engine performs several simulation steps
138  internally, this method only output the final state.
139 
140  @param[in] action_next Updated command (2D numpy array: column vector)
141  Optional: Use the value in the internal buffer otherwise
142  @param[in] dt_desired Simulation time difference between before and after the steps.
143  Optional: Perform a single integration step otherwise
144 
145  @return Final state of the simulation (2D numpy matrix: column vector ???)
146  """
147  if (action_next is not None):
148  self.action = action_next
149  self._state = None
150  return self._engine.step(dt_desired)

153  def render(self, return_rgb_array=False, lock=None):
154  """
155  @brief Render the current state of the simulation. One can display it
156  in Gepetto-viewer or return an RGB array.
157 
158  @remark Note that it supports parallel rendering, which means that one
159  can display multiple simulations in the same Gepetto-viewer
160  processus at the same time in different tabs.
161  Note that it returning an RGB array required Gepetto-viewer.
162 
163  @param[in] return_rgb_array Updated command (2D numpy array: column vector)
164  Optional: Use the value in the internal buffer otherwise
165  @param[in] lock Unique threading.Lock for every simulation
166  Optional: Only required for parallel rendering
167 
168  @return Low-resolution rendering as an RGB array (3D numpy array)
169  """
170  rgb_array = None
171 
172  if (lock is not None):
173  lock.acquire()
174  try:
175  # Instantiate the Gepetto model and viewer client if necessary
176  if (not self.is_gepetto_available):
177  self._client, self._viewer_proc = jiminy_py.get_gepetto_client(True)
178  self._id = next(tempfile._get_candidate_names())
179  self._rb = RobotWrapper()
180  collision_model = pin.buildGeomFromUrdf(self._engine.model.pinocchio_model,
181  self._engine.model.urdf_path, [],
182  pin.GeometryType.COLLISION)
183  visual_model = pin.buildGeomFromUrdf(self._engine.model.pinocchio_model,
184  self._engine.model.urdf_path, [],
185  pin.GeometryType.VISUAL)
186  self._rb.__init__(model=self._engine.model.pinocchio_model,
187  collision_model=collision_model,
188  visual_model=visual_model)
189  self.is_gepetto_available = True
190 
191  # Load model in gepetto viewer if needed
192  if not self._id in self._client.gui.getSceneList():
193  self._client.gui.createSceneWithFloor(self._id)
194  self._window_id = self._client.gui.createWindow("jiminy")
195  self._client.gui.addSceneToWindow(self._id, self._window_id)
196  self._client.gui.createGroup(self._id + '/' + self._id)
197  self._client.gui.addLandmark(self._id + '/' + self._id, 0.1)
198 
199  self._rb.initDisplay("jiminy", self._id, loadModel=False)
200  self._rb.loadDisplayModel(self._id + '/' + "robot")
201 
202  self._client.gui.setCameraTransform(self._window_id,
203  [0.0, 9.0, 2e-5, 0.0, 1.0, 1.0, 0.0])
204 
205  # Update viewer
206  jiminy_py.update_gepetto_viewer(self._rb,
207  self._engine.model.pinocchio_data,
208  self._client)
209 
210  # return rgb array if needed
211  if return_rgb_array:
212  png_path = os.path.join("/tmp", self._id + ".png")
213  self._client.gui.captureFrame(self._window_id, png_path)
214  rgb_array = np.array(Image.open(png_path))[:,:,:-1]
215  os.remove(png_path)
216  except:
217  self.is_gepetto_available = False
218  self._client = None
219  self._viewer_proc = None
220  finally:
221  if (lock is not None):
222  lock.release()
223  return rgb_array
