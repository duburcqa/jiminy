15 class JiminyCartPoleEnv(RobotJiminyEnv):
16  """
17  @brief Implementation of a Gym environment for the Cartpole which is using
18  Jiminy Engine to perform physics computations and Gepetto-viewer for
19  rendering. It is a specialization of RobotJiminyGoalEnv. The Cartpole
20  is a pole attached by an un-actuated joint to a cart. The goal is to
21  prevent the pendulum from falling over by increasing and reducing the
22  cart's velocity.
23 
24  @details **OBSERVATION:**
25  Type: Box(4)
26  Num Observation Min Max
27  0 Cart Position -1.5 1.5
28  1 Cart Velocity -Inf Inf
29  2 Pole Angle -50 deg 50 deg
30  3 Pole Velocity At Tip -Inf Inf
31 
32  **ACTIONS:**
33  Type: Discrete(2)
34  Num Action
35  0 Push cart to the left
36  1 Push cart to the right
37 
38  Note that the amount the velocity that is reduced or increased is not
39  fixed, it depends on the angle the pole is pointing. This is because the
40  center of gravity of the pole increases the amount of energy needed to
41  move the cart underneath it.
42 
43  **REWARD:**
44  Reward is 1 for every step taken, including the termination step.
45  move the cart underneath it.
46 
47  **STARTING STATE:**
48  All observations are assigned a uniform random value in [-0.05..0.05]
49 
50  **EPISODE TERMINATION:**
51  If any of these conditions is satisfied:
52  - Pole Angle is more than 25 degrees Cart
53  - Position is more than 75 cm
54  - Episode length is greater than 200
55 
56  **SOLVED REQUIREMENTS:**
57  Considered solved when the average reward is greater than or equal to
58  195.0 over 100 consecutive trials.
59  """

157  """
158  @brief Run a simulation step for a given.
159 
160  @param[in] action The action to perform (in the action space rather than
161  the original torque space).
162 
163  @return The next state, the reward, the status of the simulation (done or not),
164  and an empty dictionary for compatibility with Gym OpenAI.
165  """
166  assert self.action_space.contains(action), "%r (%s) invalid"%(action, type(action))
167 
168  # Bypass 'action' setter and use direct assignment to max out the performances
169  if action == 1:
170  self.engine_py._action[0] = self.force_mag
171  else:
172  self.engine_py._action[0] = -self.force_mag
173  self.engine_py.step(dt_desired=self.dt)
174  self.state = self.engine_py.state
175 
176  # Check the terminal condition and compute reward
177  done = self._is_success()
178  reward = 0.0
179  if not done:
180  reward += 1.0
181  elif self.steps_beyond_done is None:
182  self.steps_beyond_done = 0
183  reward += 1.0
184  else:
185  if self.steps_beyond_done == 0:
186  logger.warn("You are calling 'step()' even though this environment has already returned done = True. You should always call 'reset()' once you receive 'done = True' -- any further steps are undefined behavior.")
187  self.steps_beyond_done += 1
188
