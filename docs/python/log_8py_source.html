9 def extract_state_from_simulation_log(log_header, log_data, urdf_path, jiminy_model=None, has_freeflyer=True):
10  """
11  @brief Extract a trajectory object using from raw simulation data.
12 
13  @details Extract time, joint positions and velocities evolution from log.
14  .
15  @remark Note that the quaternion angular velocity vectors are expressed
16  it body frame rather than world frame.
17 
18  @param[in] log_header List of field names
19  @param[in] log_data Data logged (2D numpy array: row = time, column = data)
20  @param[in] urdf_path Full path of the URDF file
21  @param[in] jiminy_model Jiminy model. Optional: None if omitted
22  @param[in] has_freeflyer Whether the model has a freeflyer
23 
24  @return Trajectory dictionary. The actual trajectory corresponds to
25  the field "evolution_robot" and it is a list of State object.
26  The other fields are additional information.
27  """
28  t = log_data[:,log_header.index('Global.Time')]
29  qe = log_data[:,np.array(['currentFreeflyerPosition' in field
30  or 'currentPosition' in field for field in log_header])].T
31  dqe = log_data[:,np.array(['currentFreeflyerVelocity' in field
32  or 'currentVelocity' in field for field in log_header])].T
33  ddqe = log_data[:,np.array(['currentFreeflyerAcceleration' in field
34  or 'currentAcceleration' in field for field in log_header])].T
35 
36  # Create state sequence
37  evolution_robot = []
38  for i in range(len(t)):
39  evolution_robot.append(State(qe[:,[i]], dqe[:,[i]], ddqe[:,[i]], t[i]))
40 
41  return {"evolution_robot": evolution_robot,
42  "urdf": urdf_path,
43  "has_freeflyer": has_freeflyer,
44  "jiminy_model": jiminy_model}
