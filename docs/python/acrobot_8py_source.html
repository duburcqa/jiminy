15 class JiminyAcrobotGoalEnv(RobotJiminyGoalEnv):
16  """
17  @brief Implementation of a Gym environment for the Acrobot which is using
18  Jiminy Engine to perform physics computations and Gepetto-viewer for
19  rendering. It is a specialization of RobotJiminyGoalEnv. The acrobot
20  is a 2-link pendulum with only the second joint actuated. Initially,
21  both links point downwards. The goal is to swing the end-effector at
22  a height at least the length of one link above the base. Both links
23  can swing freely and can pass by each other, i.e. they don't collide
24  when they have the same angle.
25 
26  @details **STATE:**
27  The state consists of the sin() and cos() of the two rotational joint
28  angles and the joint angular velocities :
29  [cos(theta1) sin(theta1) cos(theta2) sin(theta2) thetaDot1 thetaDot2].
30  For the first link, an angle of 0 corresponds to the link pointing
31  downwards. The angle of the second link is relative to the angle of
32  the first link. An angle of 0 corresponds to having the same angle
33  between the two links. A state of [1, 0, 1, 0, ..., ...] means that
34  both links point downwards.
35 
36  **ACTIONS:**
37  The action is either applying +1, 0 or -1 torque on the joint between
38  the two pendulum links.
39 
40  @see R. Sutton: Generalization in Reinforcement Learning:
41  Successful Examples Using Sparse Coarse Coding (NIPS 1996)
42  @see R. Sutton and A. G. Barto:
43  Reinforcement learning: An introduction.
44  Cambridge: MIT press, 1998.
45  """

178  def _sample_goal(self):
179  """
180  @brief Samples a new goal and returns it.
181 
182  @details The goal is randomly sampled using a uniform
183  distribution between `0.2*self._tipPosZMax` and
184  `0.98*self._tipPosZMax`.
185 
186  @remark `self._tipPosZMax` can be overwritten to tweak the
187  difficulty of the problem. By default, it is the
188  maximum high of the highest joint that can be reached
189  by the pendulum.

196  def step(self, a):
197  """
198  @brief Run a simulation step for a given.
199 
200  @param[in] a The action to perform (in the action space rather than
201  the original torque space).
202 
203  @return The next observation, the reward, the status of the simulation
204  (done or not), and a dictionary of extra information
205  """
206  if self.continuous:
207  torque = a
208  else:
209  torque = self.AVAIL_TORQUE[a] * self.torque_mag
210 
211  # Add noise to the force action
212  if self.torque_noise_max > 0:
213  torque += self.np_random.uniform(-self.torque_noise_max, self.torque_noise_max)
214 
215  # Bypass 'self.engine_py.step' method and use direct assignment to max out the performances
216  self.engine_py._action[0] = torque
217  self.engine_py.step(dt_desired=self.dt)
218  self.state = self.engine_py.state
219 
220  # Get information
221  info, obs = self._get_info()
222  done = info['is_success']
223 
224  # Make sure the simulation is not already over
225  if done:
226  if self.steps_beyond_done is None:
227  self.steps_beyond_done = 0
228  else:
229  if self.steps_beyond_done == 0:
230  logger.warn("You are calling 'step()' even though this environment has already returned done = True. You should always call 'reset()' once you receive 'done = True' -- any further steps are undefined behavior.")
231  self.steps_beyond_done += 1
232 
233  # Compute the reward
234  reward = self.compute_reward(obs['achieved_goal'], self.goal, info)
235 
236  return obs, reward, done, info

264  """
265  @brief Compute the reward at the current state.
266 
267  @param[in] achieved_goal Currently achieved goal
268  @param[in] desired_goal Desired goal

270 
271  @return The computed reward.
272  """

278  # Recompute the info if not available
279  done = self._is_success(achieved_goal, desired_goal)
280

314  def _get_obs(self):
315  """
316  @brief Get the current observation based on the current state of the model.
317 
318  @remark This is a hidden function that is not listed as part of the
319  member methods of the class. It is not intended to be called
320  manually.
321 
322  @return Dictionary with the current observation, achieved goal,
323  and desired goal.
324  """
325  theta1, theta2, theta1_dot, theta2_dot = self.state
326  theta1_dot = min(max(theta1_dot, -self.MAX_VEL_1), self.MAX_VEL_1)
327  theta2_dot = min(max(theta2_dot, -self.MAX_VEL_2), self.MAX_VEL_2)
328  observation = np.array([cos(theta1 + pi),
329  sin(theta1 + pi),
330  cos(theta2 + pi),
331  sin(theta2 + pi),
332  theta1_dot,
333  theta2_dot])
334 
335  achieved_goal = self._get_achieved_goal()
336  return {
337  'observation': observation,
338  'achieved_goal': achieved_goal.copy(),
339  'desired_goal': self.goal.copy(),
340  }
