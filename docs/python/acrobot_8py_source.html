15 class JiminyAcrobotGoalEnv(RobotJiminyGoalEnv):
16  """
17  @brief Implementation of a Gym environment for the Acrobot which is using
18  Jiminy Engine to perform physics computations and Gepetto-viewer for
19  rendering. It is a specialization of RobotJiminyGoalEnv. The acrobot
20  is a 2-link pendulum with only the second joint actuated. Initially,
21  both links point downwards. The goal is to swing the end-effector at
22  a height at least the length of one link above the base. Both links
23  can swing freely and can pass by each other, i.e. they don't collide
24  when they have the same angle.
25 
26  @details **STATE:**
27  The state consists of the sin() and cos() of the two rotational joint
28  angles and the joint angular velocities :
29  [cos(theta1) sin(theta1) cos(theta2) sin(theta2) thetaDot1 thetaDot2].
30  For the first link, an angle of 0 corresponds to the link pointing
31  downwards. The angle of the second link is relative to the angle of
32  the first link. An angle of 0 corresponds to having the same angle
33  between the two links. A state of [1, 0, 1, 0, ..., ...] means that
34  both links point downwards.
35 
36  **ACTIONS:**
37  The action is either applying +1, 0 or -1 torque on the joint between
38  the two pendulum links.
39 
40  @see R. Sutton: Generalization in Reinforcement Learning:
41  Successful Examples Using Sparse Coarse Coding (NIPS 1996)
42  @see R. Sutton and A. G. Barto:
43  Reinforcement learning: An introduction.
44  Cambridge: MIT press, 1998.
45  """

177  def _sample_goal(self):
178  """
179  @brief Samples a new goal and returns it.
180 
181  @details The goal is randomly sampled using a uniform
182  distribution between `0.2*self._tipPosZMax` and
183  `0.98*self._tipPosZMax`.
184 
185  @remark `self._tipPosZMax` can be overwritten to tweak the

195  def step(self, a):
196  """
197  @brief Run a simulation step for a given.
198 
199  @param[in] a The action to perform (in the action space rather than
200  the original torque space).
201 
202  @return The next observation, the reward, the status of the simulation
203  (done or not), and a dictionary of extra information
204  """
205  if self.continuous:
206  torque = a
207  else:
208  torque = self.AVAIL_TORQUE[a] * self.torque_mag
209 
210  # Add noise to the force action
211  if self.torque_noise_max > 0:
212  torque += self.np_random.uniform(-self.torque_noise_max, self.torque_noise_max)
213 
214  # Bypass 'self.engine_py.step' method and use direct assignment to max out the performances
215  self.engine_py._action[0] = torque
216  self.engine_py.step(dt_desired=self.dt)
217  self.state = self.engine_py.state
218 
219  # Get information
220  info, obs = self._get_info()
221  done = info['is_success']
222 
223  # Make sure the simulation is not already over
224  if done:
225  if self.steps_beyond_done is None:
226  self.steps_beyond_done = 0
227  else:
228  if self.steps_beyond_done == 0:
229  logger.warn("You are calling 'step()' even though this environment has already returned done = True. You should always call 'reset()' once you receive 'done = True' -- any further steps are undefined behavior.")
230  self.steps_beyond_done += 1
231 
232  # Compute the reward
233  reward = self.compute_reward(obs['achieved_goal'], self.goal, info)
234 
235  return obs, reward, done, info

313  def _get_obs(self):
314  """
315  @brief Get the current observation based on the current state of the model.
316 
317  @remark This is a hidden function that is not listed as part of the
318  member methods of the class. It is not intended to be called
319  manually.
320 
321  @return Dictionary with the current observation, achieved goal,
322  and desired goal.
323  """
324  theta1, theta2, theta1_dot, theta2_dot = self.state
325  theta1_dot = min(max(theta1_dot, -self.MAX_VEL_1), self.MAX_VEL_1)
326  theta2_dot = min(max(theta2_dot, -self.MAX_VEL_2), self.MAX_VEL_2)
327  observation = np.array([cos(theta1 + pi),
328  sin(theta1 + pi),
329  cos(theta2 + pi),
330  sin(theta2 + pi),
331  theta1_dot,
332  theta2_dot])
333 
334  achieved_goal = self._get_achieved_goal()
335  return {
336  'observation': observation,
337  'achieved_goal': achieved_goal.copy(),
338  'desired_goal': self.goal.copy(),
339  }
