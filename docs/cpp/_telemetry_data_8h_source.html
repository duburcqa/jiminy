233  template <typename T>
234  result_t TelemetryData::internalRegisterVariable(struct memHeader * header,
235  std::string const & variableName,
236  T * & positionInBufferOut)
237  {
238  char_t * const memAddress = reinterpret_cast<char_t*>(header);
239 
240  // Check in local cache before.
241  auto entry = entriesMap_.find(variableName);
242  if (entry != entriesMap_.end())
243  {
244  positionInBufferOut = static_cast<T*>(entry->second);
245  return result_t::SUCCESS;
246  }
247 
248  // Check in shared memory.
249  int32_t positionInBuffer = findEntry(header, variableName);
250  if (positionInBuffer != -1)
251  {
252  char_t * address = memAddress + header->startDataSection + sizeof(T) * static_cast<uint32_t>(positionInBuffer);
253  entriesMap_[variableName] = static_cast<void*>(address);
254  positionInBufferOut = static_cast<T*>(entriesMap_[variableName]);
255  return result_t::SUCCESS;
256  }
257 
258  if (not header->isRegisteringAvailable)
259  {
260  std::cout << "Error - TelemetryData::updateValue - Entry not found: register it if possible." << std::endl;
261  return result_t::ERROR_GENERIC;
262  }
263 
264  if ((header->nextFreeNameOffset + static_cast<int64_t>(variableName.size()) + 1) >= header->startDataSection)
265  {
266  std::cout << "Error - TelemetryData::updateValue - TODO" << std::endl; //TODO: write appropriate error message
267  return result_t::ERROR_GENERIC;
268  }
269 
270  char_t * const namePos = memAddress + header->nextFreeNameOffset; // Compute record address
271  memcpy(namePos, variableName.data(), variableName.size());
272  header->nextFreeNameOffset += variableName.size();
273  header->nextFreeNameOffset += 1U; // Null-terminated.
274 
275 
276  char_t * const dataPos = memAddress + header->nextFreeDataOffset;
277  entriesMap_[variableName] = static_cast<void*>(dataPos);
278  positionInBufferOut = static_cast<T*>(entriesMap_[variableName]);
279  header->nextFreeDataOffset += sizeof(T);
280 
281  return result_t::SUCCESS;
282  }
