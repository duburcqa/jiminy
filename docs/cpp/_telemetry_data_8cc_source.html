79  result_t TelemetryData::registerConstant(std::string const & variableNameIn,
80  std::string const & constantValueIn)
81  {
82  // Targeted shared memory.
83  struct memHeader * const header = constantsHeader_;
84  char_t * const memAddress = reinterpret_cast<char_t *>(header);
85 
86  if (!header->isRegisteringAvailable)
87  {
88  std::cout << "result_t - TelemetryData::registerConstant - Registration is locked." << std::endl;
89  return result_t::ERROR_GENERIC;
90  }
91 
92  std::string const fullConstant = variableNameIn + "=" + constantValueIn;
93  if ((header->nextFreeNameOffset + static_cast<int64_t>(fullConstant.size()) + 1) >= header->startDataSection)
94  {
95  std::cout << "result_t - TelemetryData::registerConstant - Maximum number of registration exceeded." << std::endl;
96  return result_t::ERROR_GENERIC;
97  }
98 
99  if (findEntry(header, fullConstant) != -1)
100  {
101  std::cout << "result_t - TelemetryData::registerConstant - A constant with this name was already registered." << std::endl;
102  return result_t::ERROR_GENERIC;
103  }
104 
105  char_t * const namePos = memAddress + header->nextFreeNameOffset; // Compute record address
106  memcpy(namePos, fullConstant.data(), fullConstant.size());
107  header->nextFreeNameOffset += fullConstant.size();
108  header->nextFreeNameOffset += 1U; // Null-terminated.
109 
110  return result_t::SUCCESS;;
111  }

113  int32_t TelemetryData::findEntry(struct memHeader * header,
114  std::string const & name)
115  {
116  char_t * const memAddress = reinterpret_cast<char_t *>(header);
117 
118  int32_t position = 0;
119  int64_t currentPos = header->startNameSection;
120  while (currentPos < header->nextFreeNameOffset)
121  {
122  std::string const entry(memAddress + currentPos);
123  if (entry == name)
124  {
125  return position;

127 
128  currentPos += entry.size();
129  currentPos += 1U; // Null-terminated
130 
131  position++;
132  }

137  void TelemetryData::formatHeader(std::vector<char_t> & header) const
138  {
139  // Lock registering.
140  constantsHeader_->isRegisteringAvailable = false;
141  integersHeader_->isRegisteringAvailable = false;
142  floatsHeader_->isRegisteringAvailable = false;
143 
144  header.clear();
145  header.reserve(64 * 1024);
146 
147  // Record format version.
148  header.resize(sizeof(int32_t)); // version.
149  header[0] = ((TELEMETRY_VERSION & 0x000000ff) >> 0);
150  header[1] = ((TELEMETRY_VERSION & 0x0000ff00) >> 8);
151  header[2] = ((TELEMETRY_VERSION & 0x00ff0000) >> 16);
152  header[3] = ((TELEMETRY_VERSION & 0xff000000) >> 24);
153 
154  // Record constants.
155  header.insert(header.end(), START_CONSTANTS.data(), START_CONSTANTS.data() + START_CONSTANTS.size());
156  header.push_back('\0');
157  char_t const* startConstants = reinterpret_cast<char_t*>(constantsHeader_) + constantsHeader_->startNameSection;
158  char_t const* stopConstants = reinterpret_cast<char_t*>(constantsHeader_) + constantsHeader_->nextFreeNameOffset;
159  header.insert(header.end(), startConstants, stopConstants);
160 
161  // Record entries numbers.
162  std::string entriesNumbers;
163  entriesNumbers += NUM_INTS;
164  entriesNumbers += std::to_string((integersHeader_->nextFreeDataOffset - integersHeader_->startDataSection) /
165  static_cast<int64_t>(sizeof(int32_t)) + 1); // +1 because we add Global.Time
166  entriesNumbers += '\0';
167  entriesNumbers += NUM_FLOATS;
168  entriesNumbers += std::to_string((floatsHeader_->nextFreeDataOffset - floatsHeader_->startDataSection) /
169  static_cast<int64_t>(sizeof(float32_t)));
170  entriesNumbers += '\0';
171  header.insert(header.end(), entriesNumbers.data(), entriesNumbers.data() + entriesNumbers.size());
172 
173  // Record header - GLobal.Time - integers, floats.
174  header.insert(header.end(), START_COLUMNS.data(), START_COLUMNS.data() + START_COLUMNS.size());
175  header.push_back('\0');
176 
177  header.insert(header.end(), GLOBAL_TIME.data(), GLOBAL_TIME.data() + GLOBAL_TIME.size());
178  header.push_back('\0');
179 
180  char_t const* startIntegersHeader = reinterpret_cast<char_t*>(integersHeader_) + integersHeader_->startNameSection;
181  char_t const* stopIntegersHeader = reinterpret_cast<char_t*>(integersHeader_) + integersHeader_->nextFreeNameOffset;
182  header.insert(header.end(), startIntegersHeader, stopIntegersHeader);
183 
184  char_t const* startFloatsHeader = reinterpret_cast<char_t*>(floatsHeader_) + floatsHeader_->startNameSection;
185  char_t const* stopFloatsHeader = reinterpret_cast<char_t*>(floatsHeader_) + floatsHeader_->nextFreeNameOffset;
186  header.insert(header.end(), startFloatsHeader, stopFloatsHeader);
187 
188  // Start data section.
189  header.insert(header.end(), START_DATA.data(), START_DATA.data() + START_DATA.size());
