557  bp::object & internalDynamicsPy,

559  {
560  /* 'co_varnames' is used along with 'co_argcount' to avoid accounting for
561  the 'self' argument in case of class method handle.
562  Note that it does not support Python *args and **kwargs. In such a case,
563  specify 'numSensorTypes' argument instead. */
564 
565  int32_t N;
566  if (numSensorTypes < 0)
567  {

570  std::size_t argsNumPy = boost::python::extract<std::size_t>(
571  commandPy.attr("func_code").attr("co_argcount"));
572  N = argsNumPy - 4 - (argsFirstPy == std::string("self"));
573  }
574  else
575  {
576  N = numSensorTypes;
577  }
578  assert(N <= PYTHON_CONTROLLER_FUNCTOR_MAX_SENSOR_TYPES || N < 0);

581  std::cout << "The functors in argument are ill-defined." << std::endl;

587  }
