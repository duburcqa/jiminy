18  /// Possible modes for a device (their availability depend of the concrete device).
19  enum OpenMode
20  {
21  NOT_OPEN = 0x000, ///< Device is not opened.
22  READ_ONLY = 0x001, ///< Read only mode.
23  WRITE_ONLY = 0x002, ///< Write only mode.
24  READ_WRITE = 0x004, ///< Read/Write mode (WARNING: (READ_ONLY | WRITE_ONLY) != READ_WRITE)
25  APPEND = 0x008, ///< Open the device in append mode.
26  TRUNCATE = 0x010, ///< Truncate the device at opening.
27  UNBUFFERED = 0x020, ///< Do not use intermediate buffer if possible.
28  NEW_ONLY = 0x040, ///< Create the device at opening, fail if the device already exists.
29  EXISTING_ONLY = 0x080, ///< Do not create the device if it does not exists.
30  NON_BLOCKING = 0x100, ///< Open the device in non blocking mode.
31  SYNC = 0x200, ///< Open the device in sync mode (ensure that write are finished at return).
32  };
